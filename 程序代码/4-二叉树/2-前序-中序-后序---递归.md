    树的遍历：
    
    1、先序遍历： 根--左--右  【需要先把根遍历完毕，然后一直遍历左子树，再遍历右子树】
    2、中序遍历： 左--根--右
    3、后序遍历： 左--右--根


    def ListNode(val,left,right):

    class Solution:
    def preorderTraversal(self, root: TreeNode):
        #为什么在方法中还需要一个方法呢，因为每次添加时候 需要对列表进行置空
        def preorder(root: TreeNode):
            if not root:
                return
            res.append(root.val)
            preorder(root.left)            
            preorder(root.right)
            
        res = []
        preorder(root)
        return res
    
    
    #中序遍历
    class Solution:
    def inorderTraversal(self, root: TreeNode):
		
        def inorder(root: TreeNode):
            if not root:
                return
            inorder(root.left)
            res.append(root.val)
            inorder(root.right)
        
        res = []
        inorder(root)
        return res

   class Solution:

        def postorder(root: TreeNode):
            if not root:

            postorder(root.left)
            res.append(root.val)
            postorder(root.right)
        
        res = []

        return res
