
       中序遍历（非递归）
       def midSearch(root):
            '''
            非递归
            '''
            res = []#待返回的数组
            stack = []#存入节点的数组
            if not root:
                return []
            while root or stack:
                while root:
                    stack.append(root)
                    root = root.left
                if stack:
                    a =stack.pop()
                    root = a.right
                    res.append(a.val)
            return res
