    一个数组coins,表示不同面值的硬币；现在再给一个数组，希望通过这个硬币输出这个值所用到的最小的硬币数；
    如果没有则返回-1
    
    
    class Solution:
        def coinChange(self, coins: List[int], amount: int) -> int:
            dp = [inf]*(amount+1)
            dp[0] = 0
            for i in range(1, amount+1):
                for coin in coins:
                    if i-coin>=0:
                        dp[i] = min(dp[i], dp[i-coin]+1)
            return dp[-1] if dp[-1]!=inf else -1
